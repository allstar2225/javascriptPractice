- if문 -

+ 표현식의 결과가 참일때 실행, 거짓이면 아무것도 실행하지 않는 조건문

        var x = 10, y = 20;

        if (x == y) {
            document.write("x는 y와 같다");
        }
        if (x < y) {
            document.write("x는 y보다 작다")
        }
        if (x > y) {
            document.write("x는 y보다 크다")
        }

- else 문 -

+ if와 같이 사용되며 거짓의 값을 출력할때 사용

        var x = 10, y = 20;

        if (x == y) {
            document.write("x는 y와 같다");
        } else {
            if (x <= y) 
                document.write("x는 y보다 작습니다");
            else
                document.write("x가 y보다 큽니다");
        }

- else if 문 -

+ 중첩된 if문을 좀더 간결하게 표현할때 사용

        var x = 10, y = 20;

        if (x == y) {
            document.write("x는 y와 같다");
        } else if(x < y) {
            document.write("x는 y보다 작습니다");
        } else {
            document.write("x가 y보다 큽니다.");
        }

- switch 문 -

+ 중첩된 if문을 좀더 간결하게 표현할때 사용

        var x = 10;
        switch (typeof x) {
            case "number":
                document.write("변수 x는 숫자 타입이다.");
                break;
            case "string":
                document.write("변수 x는 문자열 타입이다.");
                break;
            case "object":
                document.write("변수 x는 객체 타입이다.");
                break;
            default:
                document.write("변수 x가 뭔지 모르겟다.");
                break;
        }

- switch 문 -

+ case형식으로 주어진 조건에 따라 위치에서 값이 수행되도록 한다

        var x = 10;
        switch (typeof x) {
            case "number":
                document.write("변수 x는 숫자 타입이다.");
                break;
            case "string":
                document.write("변수 x는 문자열 타입이다.");
                break;
            case "object":
                document.write("변수 x는 객체 타입이다.");
                break;
            default:
                document.write("변수 x가 뭔지 모르겟다.");
                break;
        }

- switch 응용 -

+ switch문의 응용 방식

        var day = new Date().getDate(); //오늘이 몇일인지
        switch (day) {
            case 1: // 월요일        
            case 2: // 화요일
            case 3: // 수요일
            case 4: // 목요일
            default: // 0부터 6까지 안나올때
            document.write("주말 언제오니");
            break;
            case 5: // 금요일 일때
            document.write("불금입니다");
            break;
            case 6: // 토요일 일때
            case 0: // 일요일
            document.write("호울리쉿.. 일요일..");
            break;
        }
