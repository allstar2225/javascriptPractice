- 배열 -

+ 배열 생성의 예

        // 첫번쨰
        var arrList = [1, true, "java"]; // 배열 리터럴을 이용
        var arrObj = Array(1, true, "java"); // Array 객체를 이용
        var arrNewObj = new Array(1, true, "java") // new 연산자를 이용한 Array 객체 생성

        document.write(arrList + " : 리터럴 방식");
        document.write(arrObj + " : 객체 방식");
        document.write(arrNewObj + " : new 연산자 방식");

        // 두번쨰
        var arr = ["java"]; // 하나 뿐인 요소를 배열에 생성
        var element = arr[0]; // 배열의 첫번째 요소를 읽어 element에 대입.
        arr[1] = 10; // 배열에 10 숫자를 대입. 길이가 1에서 2로 증가
        arr[2] = element; // 배열에 element값을 대입. 길이가 2에서 3으로 증가

        document.write("배열 arr의 요소에는 [ " + arr + " ]가 들어 있습니다");
        document.write("배열 arr의 길이는 [ " + arr.length + " ]가 들어 있습니다.");

        delete arr[2]; // 배열의 3번째 요소 삭제. 배열의 길이는 유지됨.
        
        document.write("배열의 arr 요소에는 ["+ arr +"] 가 있습니다.")
        document.write("배열 arr의 길이는 [ " + arr.length + " ]가 들어 있습니다.");

        // 세번쨰
        var arr = [1, true, "java"]; // 리터럴 방식

        arr.push("script"); // push() 메소드 사용.
        document.write("세번째")
        document.write("# push()메소드를 사용함");
        document.write(arr);

        arr[arr.length] = 100; // length 프로퍼티 사용.
        document.write("# length 프로퍼티를 사용함";
        document.write(arr);

        arr[10] = "자바스크립트"; // 특정 인덱스 추가.
        document.write("# 특정인덱스에 추가. 그외에는 undefined가 뜸);
        document.write(arr);
        document.write(arr[6]);

        // 네번쨰
        document.write("네번째
")
        var arr = [1, true, "java"]; // 리터럴 방식
        
        var result = "";
        
        for (var idx in arr){
            result += "";
        }
        result += "
" + arr[idx] + "
";

        document.write(result);
