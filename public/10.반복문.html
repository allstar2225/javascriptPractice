- while문 -

+ 특정 조건을 만족할때까지 계속 실행

        var i = 1;
        while (i < 10) { // i 가 10보다 작을때 while문 실행
            document.write("반복숫자 : " + i );
            i++; // 한바퀴마다 i에 1씩 더한다
        }

- do / while문 -

+ 먼저 루프를 한번 실행하고 표현식을 검사하기에 결과에 상관없이 일단 한번은 실행함

        var i = 1, j = 1;
        while (i > 3) { // i가 1이기에 실행되지 않는다
            document.write(" i : " + i++);
        }
        do { // 이 부분은 무조건 실행이 됨
            document.write("do 부분이다.");
        } while ( j > 3); // j가 3보다 작으므로 실행하지 않는다

- for문 -

+ while과 달리 초기식,표현식,증감식을 모두 포함. 그리하여 while보다는 좀더 간결하게 반복문을 표현한다.

        for (var i = 0; i < 10; i++){
            document.write("i : " + i);
        }

- for / in 문 -

+ for와 달리 해당 객체의 모든 프로퍼티를 열거하여 순회 할수 있도록 함.

        var obj = { name: "조현수", age: 20 };
        for (var i in obj) {
            document.write( i);
        }

- for / of 문 -

+ 반복할수 있는 객체를 순회하도록 해준다. 반복객체로는 Array, Map, Set, arguments가 있다.

        - 첫번쨰 -
        var arr = [3,4,5];
        for( var i = 0; i < arr.length; i++) { // 일반적인 for문
            document.write(arr[i]);
        }
        for ( var value of arr) { // of를 사용하여 위와 같은 같은 결과를 얻을수 있다.
            document.write(value);
        }

        - 두번째 -
        var arr2 = new Set([1, 1, 2, 2, 3, 3]);

        for (var value of arr2) {
            document.write("
" + value + "
 ")
        }
